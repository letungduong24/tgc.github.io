document.addEventListener('DOMContentLoaded', function () { const uploadPNG = document.getElementById('uploadPNG'); const canvas = document.getElementById('canvas'); const ctx = canvas.getContext('2d'); const downloadButton = document.getElementById('downloadButton'); // Nút điều khiển const zoomInButton = document.getElementById('zoomIn'); const zoomOutButton = document.getElementById('zoomOut'); const rotateLeftButton = document.getElementById('rotateLeft'); const rotateRightButton = document.getElementById('rotateRight'); let pngImage = new Image(); let frameImage = new Image(); let isPNGUploaded = false; // Các biến để điều khiển ảnh PNG let scale = 1; let rotation = 0; let offsetX = 0; let offsetY = 0; let isDragging = false; let dragStartX, dragStartY; // Biến lưu ảnh canvas dưới dạng Blob let cachedImageBlob = null; // Tải ảnh khung từ đường dẫn (có thêm cache) frameImage.src = 'Khungavt1.png?v=1'; frameImage.onload = function () { initializeCanvas(); }; // Thiết lập kích thước canvas theo kích thước khung function initializeCanvas() { // Điều chỉnh kích thước canvas theo độ phân giải của thiết bị const scaleFactor = window.devicePixelRatio || 1; canvas.width = frameImage.width * scaleFactor; canvas.height = frameImage.height * scaleFactor; ctx.scale(scaleFactor, scaleFactor); // Tăng cường độ phân giải canvas để phù hợp với màn hình di động drawImages(); } // Khi người dùng tải ảnh PNG đã xóa nền lên uploadPNG.addEventListener('change', function (event) { const file = event.target.files[0]; if (file && ['image/png', 'image/jpeg', 'image/jpg'].includes(file.type)) { const reader = new FileReader(); reader.onload = function (e) { pngImage.src = e.target.result; pngImage.onload = () => { isPNGUploaded = true; // Tự động căn chỉnh ảnh vào khung const scaleX = canvas.width / pngImage.width; const scaleY = canvas.height / pngImage.height; scale = Math.min(scaleX, scaleY); offsetX = (canvas.width - pngImage.width * scale) / 2; offsetY = (canvas.height - pngImage.height * scale) / 2; drawImages(); downloadButton.disabled = false; }; }; reader.readAsDataURL(file); } else { alert('Vui lòng chọn file PNG, JPG hoặc JPEG.'); } }); // Vẽ ảnh PNG và ảnh khung function drawImages() { ctx.clearRect(0, 0, canvas.width, canvas.height); // Nếu ảnh PNG đã được tải lên if (isPNGUploaded) { ctx.save(); ctx.translate(offsetX + pngImage.width * scale / 2, offsetY + pngImage.height * scale / 2); ctx.rotate((rotation * Math.PI) / 180); ctx.scale(scale, scale); ctx.drawImage(pngImage, -pngImage.width / 2, -pngImage.height / 2); ctx.restore(); } // Vẽ ảnh khung ctx.drawImage(frameImage, 0, 0, canvas.width, canvas.height); // Lưu ảnh canvas dưới dạng Blob khi có thay đổi canvas.toBlob(function(blob) { cachedImageBlob = blob; }, 'image/png'); } // Zoom và Xoay ảnh zoomInButton.addEventListener('click', () => { scale += 0.1; drawImages(); }); zoomOutButton.addEventListener('click', () => { if (scale > 0.1) { scale -= 0.1; drawImages(); } }); rotateLeftButton.addEventListener('click', () => { rotation -= 15; drawImages(); }); rotateRightButton.addEventListener('click', () => { rotation += 15; drawImages(); }); // Tải xuống ảnh (sử dụng Blob) downloadButton.addEventListener('click', () => { if (cachedImageBlob) { const link = document.createElement('a'); link.href = URL.createObjectURL(cachedImageBlob); link.download = 'HaoQuang.png'; link.click(); } else { alert('Chưa có ảnh để tải xuống!'); } }); // Xử lý di chuyển ảnh canvas.addEventListener('mousedown', function (e) { if (isPNGUploaded) { isDragging = true; dragStartX = e.offsetX - offsetX; dragStartY = e.offsetY - offsetY; } }); canvas.addEventListener('mousemove', function (e) { if (isDragging) { offsetX = e.offsetX - dragStartX; offsetY = e.offsetY - dragStartY; drawImages(); } }); canvas.addEventListener('mouseup', function () { isDragging = false; }); canvas.addEventListener('mouseleave', function () { isDragging = false; }); });
