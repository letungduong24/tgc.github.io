// Xử lý kéo ảnh PNG bằng chuột canvas.addEventListener('mousedown', function (event) { isDragging = true; startX = event.offsetX - offsetX; startY = event.offsetY - offsetY; }); canvas.addEventListener('mousemove', function (event) { if (isDragging) { offsetX = event.offsetX - startX; offsetY = event.offsetY - startY; drawImages(); } }); canvas.addEventListener('mouseup', function () { isDragging = false; }); canvas.addEventListener('mouseleave', function () { isDragging = false; }); // Xử lý kéo ảnh PNG bằng cảm ứng canvas.addEventListener('touchstart', function (event) { const touch = event.touches[0]; const rect = canvas.getBoundingClientRect(); isDragging = true; startX = touch.clientX - rect.left - offsetX; startY = touch.clientY - rect.top - offsetY; }); canvas.addEventListener('touchmove', function (event) { if (isDragging) { const touch = event.touches[0]; const rect = canvas.getBoundingClientRect(); offsetX = touch.clientX - rect.left - startX; offsetY = touch.clientY - rect.top - startY; drawImages(); event.preventDefault(); // Ngăn chặn sự kiện cuộn mặc định } }); canvas.addEventListener('touchend', function () { isDragging = false; });
